Simple API Help

This page describes how to use the Simple API. The Simple API provides a JavaScript API for inserting simple maps into web pages.

See the Extended API Help page for help on the Extended API.

Basis

To use the API you should add the following HTML:

<script src="${request.route_url('apijs')}"></script>
<script>
window.onload = function() {
    // add the code here
};
</script>

To put a new map in the page you'll have to put a div element with a certain id where you want your map to be: 

<div id='map1' style='width:700px;height:400px;'></div>

A map

var map = new demo.Map({
    div: 'map1', // id of the div element to put the map in
    zoom: 5,
    center: [516000, 5885000]
});

A map with a marker on its center

var map = new demo.Map({
    div: 'map2',
    zoom: 5,
    center: [516000, 5885000]
});
map.addMarker();

A map with several custom markers

var map = new demo.Map({
    div: 'map3',
    zoom: 5,
    center: [516000, 5885000]
});
map.addMarker({
    position: [600000, 5700000],
    size: [14, 14],
    icon: '${request.static_path('demo:static/apihelp/images/info.png')}'
});
map.addMarker({
    position: [516000, 5810000],
    size: [18, 18],
    icon: '${request.static_path('demo:static/apihelp/images/essence.png')}'
});
map.addMarker({
    position: [400000, 5900000],
    size: [14, 14],
    icon: '${request.static_path('demo:static/apihelp/images/parking.png')}'
});

A map with a subset of overlays

var map = new demo.Map({
    div: 'map4',
    zoom: 12,
    center: [429600, 5405400],
    layers: ['airedejeux']
});

A map with some additional controls

var map = new demo.Map({
    div: 'map5',
    zoom: 12,
    center: [429600, 5405400],
    layers: ['airedejeux'],
    addLayerSwitcher: true,
    layerSwitcherExpanded: true,
    addMiniMap: true,
    miniMapExpanded: true,
    showCoords: true
});

Recenter the map to given coordinates


var map_ = new demo.Map({
    div: 'map6',
    zoom: 4,
    center: [516000, 5885000],
    addMiniMap: true,
    miniMapExpanded: true
});
var button1 = document.getElementById('button1');
button1.onclick = function() {
    map_.recenter([270000, 5952000], 5);
}
var button2 = document.getElementById('button2');
button2.onclick = function() {
    map_.recenter([930000, 5925000], 7);
}

Recenter the map on objects

var map = new demo.Map({
    div: 'map7',
    layers: ['airedejeux']
});
map.recenterOnObjects(
    /* the layer name */
    'airedejeux',
    /* the ids of the objects */
    ['16', '11'],
    /* whether to highlight the objects or not */
    true
);

Load data from a GPX file

var map = new demo.Map({
    div: 'map8'
});
map.addCustomLayer('gpx', 'My GPX layer', '${request.static_path('demo:static/apihelp/track0.gpx')}');
